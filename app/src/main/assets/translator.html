🔧 إعدادات Android المطلوبة
1️⃣ AndroidManifest.xml — صلاحية الإنترنت
أضف هذا قبل <application>
<uses-permission android:name="android.permission.INTERNET"/> <uses-permission android:name="android.permission.ACCESS_NETWORK_STATE"/>
2️⃣ network_security_config.xml — السماح بـ HTTPS
res/xml/network_security_config.xml
<?xml version="1.0" encoding="utf-8"?> <network-security-config> <domain-config cleartextTrafficPermitted="false"> <domain includeSubdomains="true">translate.googleapis.com</domain> <domain includeSubdomains="true">clients5.google.com</domain> <domain includeSubdomains="true">translate.google.com</domain> </domain-config> </network-security-config>
3️⃣ AndroidManifest.xml — ربط network_security_config
android:networkSecurityConfig="@xml/network_security_config"
4️⃣ WebViewActivity.java — الإعدادات الكاملة
WebSettings s = webView.getSettings(); s.setJavaScriptEnabled(true); s.setDomStorageEnabled(true); s.setAllowFileAccess(false); s.setMixedContentMode(WebSettings.MIXED_CONTENT_NEVER_ALLOW); s.setCacheMode(WebSettings.LOAD_DEFAULT); // Android Bridge webView.addJavascriptInterface( new AndroidClipboard(this), "AndroidClipboard" ); // CORS fix — override shouldInterceptRequest webView.setWebViewClient(new WebViewClient() { @Override public WebResourceResponse shouldInterceptRequest( WebView view, WebResourceRequest request) { // Let all translate.googleapis.com through return null; } });
5️⃣ AndroidClipboard Bridge — الكلاس الكامل
public class AndroidClipboard { private Context ctx; private SharedPreferences prefs; public AndroidClipboard(Context c) { ctx = c; prefs = c.getSharedPreferences("wt_prefs", Context.MODE_PRIVATE); } @JavascriptInterface public String getText() { try { ClipboardManager cm = (ClipboardManager) ctx.getSystemService(Context.CLIPBOARD_SERVICE); if (cm.hasPrimaryClip()) { ClipData.Item item = cm.getPrimaryClip().getItemAt(0); return item.getText() != null ? item.getText().toString() : ""; } } catch(Exception e) { return "ERROR:" + e.getMessage(); } return ""; } @JavascriptInterface public void setText(String text) { try { ClipboardManager cm = (ClipboardManager) ctx.getSystemService(Context.CLIPBOARD_SERVICE); cm.setPrimaryClip(ClipData.newPlainText("translation", text)); } catch(Exception e) {} } @JavascriptInterface public void saveLanguages(String from, String to) { prefs.edit().putString("from", from).putString("to", to).apply(); } @JavascriptInterface public String getFromLang() { return prefs.getString("from","auto"); } @JavascriptInterface public String getToLang() { return prefs.getString("to","ar"); } @JavascriptInterface public boolean isNetworkAvailable() { ConnectivityManager cm = (ConnectivityManager) ctx.getSystemService(Context.CONNECTIVITY_SERVICE); NetworkInfo ni = cm.getActiveNetworkInfo(); return ni != null && ni.isConnected(); } }
